Promotions & Announcements

The active banner shows at the top of ALL pages for ALL visitors immediately. If no active banner is set, the bar is hidden.

One-time setup required if you see a permission error below If you get a "row-level security" error when activating a banner, you need to run the database permission fix once in Supabase. It takes 30 seconds:
  1. Open app.supabase.com → your project
  2. Click SQL Editor in the left sidebar
  3. Click New query
  4. Paste and run the SQL below:
▶ Click to show the SQL to copy 
ALTER TABLE banners DISABLE ROW LEVEL SECURITY;
DROP POLICY IF EXISTS "allow anon read active banners" ON banners;
DROP POLICY IF EXISTS "allow authenticated full access banners" ON banners;
DROP POLICY IF EXISTS "allow authenticated delete banners" ON banners;
DROP POLICY IF EXISTS "allow all banners" ON banners;
DROP POLICY IF EXISTS "allow anon full access banners" ON banners;

ALTER TABLE newsletter_subscribers ENABLE ROW LEVEL SECURITY;
DROP POLICY IF EXISTS "allow anon insert newsletter" ON newsletter_subscribers;
DROP POLICY IF EXISTS "allow anon update newsletter" ON newsletter_subscribers;
DROP POLICY IF EXISTS "allow authenticated full access newsletter" ON newsletter_subscribers;
DROP POLICY IF EXISTS "allow admin all newsletter" ON newsletter_subscribers;
DROP POLICY IF EXISTS "allow anon upsert newsletter" ON newsletter_subscribers;
CREATE POLICY "allow anon insert newsletter" ON newsletter_subscribers FOR INSERT TO anon WITH CHECK (true);
CREATE POLICY "allow anon update newsletter" ON newsletter_subscribers FOR UPDATE TO anon USING (true) WITH CHECK (true);
CREATE POLICY "allow authenticated full access newsletter" ON newsletter_subscribers FOR ALL TO authenticated USING (true) WITH CHECK (true);

DROP POLICY IF EXISTS "allow authenticated read all profiles" ON profiles;
CREATE POLICY "allow authenticated read all profiles" ON profiles FOR SELECT TO authenticated USING (true);

-- ── ORDERS TABLE — Add missing columns ───────────────────────────
-- Run this if you see: "Could not find the 'paid_at' column"
ALTER TABLE orders ADD COLUMN IF NOT EXISTS paid_at TIMESTAMPTZ;
ALTER TABLE orders ADD COLUMN IF NOT EXISTS payment_gateway TEXT;
ALTER TABLE orders ADD COLUMN IF NOT EXISTS payment_gateway_ref TEXT;
ALTER TABLE orders ADD COLUMN IF NOT EXISTS order_source TEXT DEFAULT 'website';
ALTER TABLE orders ADD COLUMN IF NOT EXISTS notes TEXT;
ALTER TABLE profiles ADD COLUMN IF NOT EXISTS whatsapp_number TEXT;
ALTER TABLE profiles ADD COLUMN IF NOT EXISTS phone TEXT;
